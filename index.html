function loadFiles(itemId){
  try { return JSON.parse(localStorage.getItem(`files_${itemId}`) || '[]'); }
  catch(e){ return []; }
}
function saveFiles(itemId, files){
  localStorage.setItem(`files_${itemId}`, JSON.stringify(files));
}

function render(){
  const q = document.getElementById('search').value.trim().toLowerCase();
  const checkedOnly = document.getElementById('showCheckedOnly').checked;
  app.innerHTML = '';

  let total = 0, visible = 0, checked = 0, visibleSections = 0;

  sections.forEach(section => {
    const visibleItemsForSection = section.items.filter(item => {
      const storyText = buildOwnWords(section.title, item.text).join(' ').toLowerCase();
      const blob = `${section.title} ${item.stars} ${item.proof} ${item.text} ${storyText}`.toLowerCase();
      const isChecked = !!checks[item.id];
      return (!q || blob.includes(q)) && (!checkedOnly || isChecked);
    });

    total += section.items.length;
    checked += section.items.filter(i => checks[i.id]).length;

    if (!visibleItemsForSection.length) return;
    visibleSections++;
    visible += visibleItemsForSection.length;

    const sec = document.createElement('div');
    sec.className = 'section';

    const head = document.createElement('div');
    head.className = 'section-header';
    head.innerHTML = `
      <h2 class="section-title">${esc(section.title)}</h2>
      <div class="section-meta">${visibleItemsForSection.length} visible / ${section.items.length} total</div>
    `;

    const body = document.createElement('div');
    body.className = 'section-body';

    visibleItemsForSection.forEach(item => {
      const row = document.createElement('div');
      row.className = 'item';
      row.dataset.id = item.id;
      row.dataset.visible = '1';

      const isChecked = !!checks[item.id];
      const defaultStory = buildOwnWords(section.title, item.text);
      const savedStory = storyEdits[item.id] || '';
      const storySource = savedStory
        ? savedStory.split(/\n{2,}/).filter(Boolean)
        : defaultStory;

      const storyParas = storySource
        .map(p => `<p>${highlight(p, q)}</p>`)
        .join('');

      const storyEditorValue = savedStory || defaultStory.join('\n\n');

      row.innerHTML = `
        <input type="checkbox" ${isChecked ? 'checked' : ''} />
        <div class="item-main">
          <div class="item-top">
            ${item.stars ? `<span class="pill stars">${esc(item.stars)}</span>` : ''}
            ${item.proof ? `<span class="pill proof">[${esc(item.proof)}]</span>` : ''}
          </div>

          <div class="text">${highlight(item.text, q)}</div>

          <details class="story">
            <summary><span class="story-caret">▸</span><span>show my story</span></summary>
            <div class="story-box">
              <div class="story-title">In my own words</div>

              <div class="story-controls">
                <button class="story-btn" data-edit-story="${item.id}">Edit</button>
                <button class="story-btn" data-save-story="${item.id}" style="display:none;">Save</button>
                <button class="story-btn" data-cancel-story="${item.id}" style="display:none;">Cancel</button>
                <button class="story-btn danger" data-reset-story="${item.id}">Reset to base</button>
              </div>

              <div class="story-display" data-story-display="${item.id}">
                ${storyParas}
              </div>

              <div class="story-editor-wrap" data-story-editor-wrap="${item.id}">
                <textarea class="story-editor" data-story-editor="${item.id}">${esc(storyEditorValue)}</textarea>
              </div>
            </div>
          </details>

          <div class="item-files">
            <input type="file" id="file-${item.id}" class="file-input" accept=".txt,image/*,.pdf,.doc,.docx" />
            <button class="upload-btn" data-upload="${item.id}">📤 Upload file</button>
            <div data-files="${item.id}"></div>
          </div>
        </div>
      `;

      body.appendChild(row);
    });

    sec.appendChild(head);
    sec.appendChild(body);
    app.appendChild(sec);
  });

  document.getElementById('statTotal').textContent = total;
  document.getElementById('statVisible').textContent = visible;
  document.getElementById('statChecked').textContent = checked;
  document.getElementById('statSections').textContent = visibleSections;
  document.getElementById('empty').style.display = visible === 0 ? 'block' : 'none';

  document.querySelectorAll('.item input[type="checkbox"]').forEach(cb => {
    cb.addEventListener('change', e => {
      const id = e.target.closest('.item').dataset.id;
      if (e.target.checked) checks[id] = true;
      else delete checks[id];
      saveChecks(checks);
      render();
    });
  });

  document.querySelectorAll('[data-upload]').forEach(btn => {
    btn.addEventListener('click', e => {
      const id = btn.getAttribute('data-upload');
      const input = document.getElementById(`file-${id}`);
      if (input) input.click();
    });
  });

  document.querySelectorAll('.file-input').forEach(input => {
    input.addEventListener('change', e => {
      const file = e.target.files && e.target.files[0];
      if (!file) return;
      const itemId = e.target.closest('.item').dataset.id;
      const reader = new FileReader();
      reader.onload = evt => {
        const files = loadFiles(itemId);
        files.push({
          name: file.name,
          size: file.size,
          data: evt.target.result,
          timestamp: new Date().toLocaleString()
        });
        saveFiles(itemId, files);
        render();
      };
      reader.readAsDataURL(file);
    });
  });

  document.querySelectorAll('[data-files]').forEach(container => {
    const itemId = container.getAttribute('data-files');
    const files = loadFiles(itemId);
    container.innerHTML = '';

    files.forEach((file, idx) => {
      const fileDiv = document.createElement('div');
      fileDiv.className = 'file-item';
      fileDiv.innerHTML = `
        <a href="${file.data}" download="${esc(file.name)}">📎 ${esc(file.name)} (${(file.size/1024).toFixed(1)}KB)</a>
        <button class="delete-file-btn" data-delete="${itemId}" data-index="${idx}">Delete</button>
      `;
      container.appendChild(fileDiv);
    });
  });

  document.querySelectorAll('[data-delete]').forEach(btn => {
    btn.addEventListener('click', () => {
      const itemId = btn.getAttribute('data-delete');
      const idx = Number(btn.getAttribute('data-index'));
      const files = loadFiles(itemId);
      files.splice(idx, 1);
      saveFiles(itemId, files);
      render();
    });
  });

  document.querySelectorAll('[data-edit-story]').forEach(btn => {
    btn.addEventListener('click', e => {
      e.preventDefault();
      e.stopPropagation();

      const id = btn.getAttribute('data-edit-story');
      const box = btn.closest('.story-box');
      const display = box.querySelector(`[data-story-display="${id}"]`);
      const editorWrap = box.querySelector(`[data-story-editor-wrap="${id}"]`);
      const saveBtn = box.querySelector(`[data-save-story="${id}"]`);
      const cancelBtn = box.querySelector(`[data-cancel-story="${id}"]`);

      display.style.display = 'none';
      editorWrap.style.display = 'block';
      btn.style.display = 'none';
      saveBtn.style.display = 'inline-block';
      cancelBtn.style.display = 'inline-block';

      const ta = box.querySelector(`[data-story-editor="${id}"]`);
      if (ta) {
        ta.focus();
        ta.setSelectionRange(ta.value.length, ta.value.length);
      }
    });
  });

  document.querySelectorAll('[data-save-story]').forEach(btn => {
    btn.addEventListener('click', e => {
      e.preventDefault();
      e.stopPropagation();

      const id = btn.getAttribute('data-save-story');
      const box = btn.closest('.story-box');
      const ta = box.querySelector(`[data-story-editor="${id}"]`);
      const newText = ta ? ta.value.trim() : '';

      if (newText) storyEdits[id] = newText;
      else delete storyEdits[id];

      saveStoryEdits(storyEdits);
      render();
    });
  });

  document.querySelectorAll('[data-cancel-story]').forEach(btn => {
    btn.addEventListener('click', e => {
      e.preventDefault();
      e.stopPropagation();
      render();
    });
  });

  document.querySelectorAll('[data-reset-story]').forEach(btn => {
    btn.addEventListener('click', e => {
      e.preventDefault();
      e.stopPropagation();

      const id = btn.getAttribute('data-reset-story');
      delete storyEdits[id];
      saveStoryEdits(storyEdits);
      render();
    });
  });
}

document.getElementById('copyVisible').addEventListener('click', async () => {
  const out = [];
  document.querySelectorAll('.section').forEach(sectionEl => {
    const title = sectionEl.querySelector('.section-title')?.innerText?.trim();
    const items = [...sectionEl.querySelectorAll('.item')];
    if (!items.length) return;
    out.push(title);
    out.push('');
    items.forEach(item => {
      const checked = item.querySelector('input').checked ? '[x]' : '[ ]';
      const stars = item.querySelector('.stars')?.innerText || '';
      const proof = item.querySelector('.proof')?.innerText || '';
      const text = item.querySelector('.text')?.innerText || '';
      out.push(`${checked} ${stars} ${proof} ${text}`.replace(/\s+/g,' ').trim());
    });
    out.push('');
  });

  try {
    await navigator.clipboard.writeText(out.join('\n'));
    alert('Visible checklist copied.');
  } catch (e) {
    alert('Clipboard blocked. Copy manually if needed.');
  }
});
